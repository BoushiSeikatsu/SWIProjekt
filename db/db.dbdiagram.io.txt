
Table Storage {
  id integer [primary key]
  name varchar
}

Table Manufacturer {
  id integer [primary key]
  number integer
  name varchar
}

Table Product {
  id integer [primary key]
  manufacturer_id integer
  name varchar
  xdim integer
  ydim integer
  zdim integer
  
}

Table ProductStore {
  id integer [primary key]
  name varchar
  amount integer
  storages integer
  tile integer
}


Table StorageChange {
  id integer [primary key]
  user integer
  tile integer
  storage integer
  date datetime
  amount integer
  job integer
}

Table Job {
  id integer [primary key]
  name varchar
}

Table User {
  id integer [primary key]
  username varchar
  pwdhash varchar
  admin bool
}
