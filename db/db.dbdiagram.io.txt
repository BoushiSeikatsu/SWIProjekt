// Use DBML to define your database structure
// Docs: https://dbml.dbdiagram.io/docs
Table storages {
  id integer [primary key]
  name varchar
}

Table manufacturers {
  id integer [primary key]
  name varchar
}

Table tiles {
  id integer [primary key]
  name varchar
  xdim integer
  ydim integer
  manufacturer integer
  photo integer [Note: "ID of photo/link to bucket"]
}

Table stores {
  id integer [primary key]
  amount integer
  place integer
  tile integer
}

Table added {
  id integer [primary key]
  tile integer
  storage integer
  date datetime
  amount integer
}

Table jobs {
  id integer [primary key]
  name varchar
}

Table removed {
  id integer [primary key]
  tile integer
  storage integer
  date datetime
  amount integer
  job integer
}

Ref: tiles.manufacturer > manufacturers.id

Ref: stores.place > storages.id
Ref: stores.tile > tiles.id

Ref: added.tile > tiles.id
Ref: added.storage > storages.id

Ref: removed.tile > tiles.id
Ref: removed.storage > storages.id
Ref: removed.job > jobs.id
